**Constraining decisions:**
- Tiered circuit breaker (ROADMAP/PROJECT.md takes precedence over layer doc binary CB):
    drawdown < 10%  → 1.0x size (full)
    drawdown 10-20% → 0.5x size
    drawdown 20-30% → 0.25x size
    drawdown >= 30% → 0.0x size (no new positions, but this is a size_multiplier on sizing, not a hard flag)
  NOTE: The circuit_breaker_active flag fires at 30% (hard stop entirely). The tiered sizing applies BEFORE 30%.
- ATR proxy formula: log_returns.rolling(14).std() * close * 1.25 — this is computed in data pipeline; RiskManager receives the resulting ATR float, NOT raw prices
- Stop checks run EVERY 60s cycle independent of signal generation
- dump_state / load_state must persist: trailing_stops, entry_prices, portfolio_hwm, circuit_breaker_active
- Hard stop at 8% loss from entry (config: hard_stop_pct, default 0.08)
- ATR trailing multiplier (config: atr_stop_multiplier, default 2.0)
- Trailing stop only moves UP, never down
- Use only Free balance from exchange (never total wallet balance)

4. RiskManager.__init__(self, config: dict):
   - self.config = config
   - self._trailing_stops: dict[str, float] = {}
   - self._entry_prices: dict[str, float] = {}
   - self._portfolio_hwm: float = 0.0
   - self._circuit_breaker_active: bool = False

5. check_stops(self, pair: str, current_price: float, current_atr: float) -> StopCheckResult:
   - If pair not in self._entry_prices: return StopCheckResult(False, "No entry price", "")
   - entry_price = self._entry_prices[pair]
   - hard_stop_pct = self.config.get("hard_stop_pct", 0.08)
   - hard_stop_price = entry_price * (1 - hard_stop_pct)
   - If current_price <= hard_stop_price: log WARNING, return StopCheckResult(True, f"HARD STOP: price={current_price:.4f} <= stop={hard_stop_price:.4f}", "hard_pct")
   - If current_atr is not NaN and current_atr > 0:
       atr_mult = self.config.get("atr_stop_multiplier", 2.0)
       new_atr_stop = current_price - atr_mult * current_atr
       current_trail = self._trailing_stops.get(pair, hard_stop_price)
       updated_trail = max(current_trail, new_atr_stop)  # only moves up
       self._trailing_stops[pair] = updated_trail
       if current_price <= updated_trail: log WARNING, return StopCheckResult(True, f"ATR TRAILING STOP: price={current_price:.4f} <= stop={updated_trail:.4f}", "atr_trailing")
   - return StopCheckResult(False, "Within stop bounds", "")

6. record_entry(self, pair: str, entry_price: float, initial_stop: float): set self._entry_prices[pair]=entry_price, self._trailing_stops[pair]=initial_stop, log INFO.

7. record_exit(self, pair: str): pop pair from _entry_prices and _trailing_stops, log INFO.

8. get_current_stop(self, pair: str) -> Optional[float]: return self._trailing_stops.get(pair)

9. initialize_hwm(self, initial_portfolio_value: float): self._portfolio_hwm = max(self._portfolio_hwm, initial_portfolio_value)

1. _get_cb_size_multiplier(self, drawdown: float) -> float:
   - if drawdown >= 0.30: return 0.0
   - if drawdown >= 0.20: return 0.25
   - if drawdown >= 0.10: return 0.50
   - return 1.0

2. check_circuit_breaker(self, current_portfolio_value: float) -> bool:
   - If current_portfolio_value > self._portfolio_hwm: update HWM, if _circuit_breaker_active and new HWM >= old HWM: deactivate CB, log INFO.
   - If self._portfolio_hwm == 0: return False
   - drawdown = (self._portfolio_hwm - current_portfolio_value) / self._portfolio_hwm
   - cb_threshold = self.config.get("circuit_breaker_drawdown", 0.30)
   - If drawdown >= cb_threshold and not self._circuit_breaker_active: set active, log CRITICAL with drawdown%, HWM, current value.
   - Return self._circuit_breaker_active.

3. get_cb_size_multiplier(self, current_portfolio_value: float) -> float:
   - If _portfolio_hwm == 0 or _circuit_breaker_active: return 0.0 if _circuit_breaker_active else 1.0
   - drawdown = (self._portfolio_hwm - current_portfolio_value) / self._portfolio_hwm
   - return self._get_cb_size_multiplier(max(drawdown, 0.0))

4. size_new_position(self, pair: str, current_price: float, current_atr: float, free_balance_usd: float, open_positions: dict, regime_multiplier: float, signal_size_pct: float) -> SizingResult:
   - Gate: if _circuit_breaker_active return SizingResult(BLOCKED_CIRCUIT_BREAKER, 0, 0, 0, 0, "Circuit breaker active")
   - Gate: if regime_multiplier == 0.0 return SizingResult(BLOCKED_ZERO_REGIME_MULTIPLIER, 0, 0, 0, 0, "BEAR regime")
   - Gate: max_positions check. max_positions = self.config.get("max_positions", 1). Count pairs in open_positions with value > 0. If at max return SizingResult(BLOCKED_MAX_POSITIONS, ...)
   - Gate: usable_balance = free_balance_usd * 0.95. If usable_balance < 100 return SizingResult(BLOCKED_INSUFFICIENT_BALANCE, ...)
   - total_portfolio = free_balance_usd + sum(open_positions.values())
   - Get CB size multiplier: if _portfolio_hwm > 0: drawdown = max((self._portfolio_hwm - total_portfolio) / self._portfolio_hwm, 0.0); cb_mult = self._get_cb_size_multiplier(drawdown) else: cb_mult = 1.0
   - effective_multiplier = regime_multiplier * cb_mult
   - If effective_multiplier == 0.0 return SizingResult(BLOCKED_ZERO_REGIME_MULTIPLIER, 0, 0, 0, 0, "Combined multiplier is zero")
   - target_usd = total_portfolio * signal_size_pct * effective_multiplier
   - max_single_pct = self.config.get("max_single_position_pct", 0.40)
   - target_usd = min(target_usd, total_portfolio * max_single_pct)
   - target_usd = min(target_usd, usable_balance)
   - quantity = target_usd / current_price
   - hard_stop = current_price * (1 - self.config.get("hard_stop_pct", 0.08))
   - atr_mult = self.config.get("atr_stop_multiplier", 2.0)
   - import math; atr_stop = (current_price - atr_mult * current_atr) if (not math.isnan(current_atr) and current_atr > 0) else hard_stop
   - initial_stop = max(hard_stop, atr_stop)
   - return SizingResult(APPROVED, quantity, target_usd, hard_stop, initial_stop, f"Approved {target_usd:.0f} USD mult={effective_multiplier:.2f}")

5. dump_state(self) -> dict: return {"trailing_stops": self._trailing_stops, "entry_prices": self._entry_prices, "portfolio_hwm": self._portfolio_hwm, "circuit_breaker_active": self._circuit_breaker_active}

6. load_state(self, state: dict):
   - self._trailing_stops = state.get("trailing_stops", {})
   - self._entry_prices = state.get("entry_prices", {})
   - self._portfolio_hwm = state.get("portfolio_hwm", 0.0)
   - self._circuit_breaker_active = state.get("circuit_breaker_active", False)
   - log INFO f"RiskManager state loaded: HWM={self._portfolio_hwm:.0f}, CB={'active' if self._circuit_breaker_active else 'inactive'}, positions={list(self._entry_prices.keys())}"
