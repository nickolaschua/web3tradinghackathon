2. close_position(self, pair: str, quantity: float, current_price: float) -> Optional[ManagedOrder]:
   - place_order(pair, "SELL", quantity, current_price, 0.0)
   - On success: remove pair from self._positions, call self.risk_manager.record_exit(pair)
   - Return the ManagedOrder result.

3. _resync_from_exchange(self, force: bool = False) -> None:
   - If not force and (time.time() - self._last_reconcile_ts) < self.RECONCILE_INTERVAL: return
   - self._last_reconcile_ts = time.time()
   - Try: balance_resp = self.client.get_balance()
   - If not balance_resp.get("Success", False): log WARNING "Reconcile: balance fetch failed", return
   - balances = balance_resp.get("Data", {})  # dict of {asset: {Free, Locked, Total}}
   - For each pair in list(self._positions.keys()):
       asset = pair.split("/")[0]  # e.g. "BTC" from "BTC/USD"
       exchange_qty_str = balances.get(asset, {}).get("Total", "0")
       exchange_qty = float(exchange_qty_str)
       local_qty = self._positions[pair].quantity
       if abs(exchange_qty - local_qty) > 1e-8:  # meaningful discrepancy
           log WARNING f"Reconcile discrepancy for {pair}: local={local_qty:.8f} exchange={exchange_qty:.8f}"
           # CRITICAL: write back to self._positions (Issue 14)
           if exchange_qty > 1e-8:
               self._positions[pair].quantity = exchange_qty
               log INFO f"Reconcile: updated {pair} position to {exchange_qty:.8f}"
           else:
               del self._positions[pair]
               self.risk_manager.record_exit(pair)
               log INFO f"Reconcile: closed {pair} (exchange shows zero balance)"
   - log DEBUG f"Reconcile complete: {len(self._positions)} positions"

4. cancel_order(self, order_id: int) -> bool:
   - Try: response = self.client.cancel_order(order_id=order_id)
   - If response.get("Success", False): log INFO f"cancel_order {order_id}: success"; if order_id in self._open_orders: self._open_orders[order_id].status = OrderStatus.CANCELED; return True
   - Else: log WARNING f"cancel_order {order_id} failed: {response}"; return False
   - Except Exception as e: log ERROR f"cancel_order exception: {e}"; return False

5. dump_state(self) -> dict:
   - Return dict with:
     "open_orders": {str(oid): {"order_id": o.order_id, "pair": o.pair, "side": o.side, "quantity": o.quantity, "submitted_at": o.submitted_at, "status": o.status.name, "fill_price": o.fill_price, "filled_quantity": o.filled_quantity} for oid, o in self._open_orders.items()}
     "positions": {pair: {"pair": p.pair, "quantity": p.quantity, "entry_price": p.entry_price, "current_value_usd": p.current_value_usd} for pair, p in self._positions.items()}

6. load_state(self, state: dict):
   - Restore self._open_orders from state["open_orders"]: for each entry, create ManagedOrder(..., status=OrderStatus[status_str])
   - Restore self._positions from state["positions"]: for each entry, create Position(...)
   - log INFO f"OrderManager state loaded: {len(self._open_orders)} orders, {len(self._positions)} positions"

7. maybe_reconcile(self) -> None: call self._resync_from_exchange(force=False)  (convenience method for main loop)
